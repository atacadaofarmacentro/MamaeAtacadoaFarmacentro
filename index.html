Evite automedicação; para dores, o Paracetamol é geralmente o mais indicado, mas consulte seu médico. Inicie o pré-natal imediatamente."; sintomas = "Náuseas matinais, sensibilidade nos seios, fadiga extrema, variação de humor e aumento da frequência urinária."; } else if (semanas <= 27) { trimestre = "2º Trimestre (Fase de Crescimento)"; dica = "Geralmente inicia-se a suplementação de Ferro e Cálcio. Mantenha a hidratação para evitar infecção urinária. Use cremes antiestrias. É a fase ideal para realizar o ultrassom morfológico."; sintomas = "Diminuição dos enjoos, surgimento da 'linha nigra' no abdômen, possíveis cãibras noturnas e os primeiros movimentos do bebê."; } else if (semanas <= 42) { trimestre = "3º Trimestre (Fase de Preparação)"; dica = "Foco em Ômega 3 (DHA) para o cérebro do bebê. Se houver contrações ritmadas ou perda de líquido, procure o hospital."; sintomas = "Inchaço nos pés e mãos, falta de ar (compressão do diafragma), azia, dor lombar e dificuldade para dormir."; } else { trimestre = "Pós-termo"; dica = "Consulte seu obstetra urgentemente para monitorar o bem-estar do bebê."; sintomas = "Ansiedade e pressão pélvica intensa."; } document.getElementById("semanas").innerText = semanas; document.getElementById("trimestre").innerText = trimestre; document.getElementById("dpp").innerText = "Parto estimado: " + parto.toLocaleDateString("pt-BR"); document.getElementById("dica").innerText = dica; document.getElementById("sintomas").innerText = sintomas; document.getElementById("resultado").classList.remove("hidden"); }
